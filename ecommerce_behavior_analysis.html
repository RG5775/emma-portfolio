WITH brand_view_purchase AS (
    SELECT
        brand,
        user_session,
        MAX(CASE WHEN event_type = 'view' THEN 1 ELSE 0 END) AS viewed,
        MAX(CASE WHEN event_type = 'purchase' THEN 1 ELSE 0 END) AS purchased
    FROM cleaned_data
    GROUP BY brand, user_session
)

SELECT 
    brand,
    COUNT(user_session) FILTER (WHERE viewed = 1) AS product_view,
    ROUND(100.00 * COUNT(user_session) FILTER (WHERE purchased = 1) / 
          NULLIF(COUNT(user_session) FILTER (WHERE viewed = 1), 0), 2) AS view_to_purchase_conversion
FROM brand_view_purchase
GROUP BY brand
ORDER BY view_to_purchase_conversion DESC;

Query Insight: This CTE-based query analyzes conversion rates across 4,202 brands, identifying which brands drive the highest view-to-purchase conversions within the same user session.
